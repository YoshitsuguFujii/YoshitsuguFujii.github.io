表題の通り  

``` ruby
def hoge
  p "hoge"
end if true

hoge # => "hoge"

def bar
  p "bar"
end if false

bar # => NameError: undefined local variable or method `bar' for main:Object

def hogefuga
 puts hoge + "fuga"
end if respond_to?(hoge)

hogefuga # => # hogefuga
```
